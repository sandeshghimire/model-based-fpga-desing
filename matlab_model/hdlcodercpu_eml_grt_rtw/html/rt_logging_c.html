  312   /* Function: rt_GetDblValueFromOverSizedData ===================================
  313    * Abstract:
  314    */
  315   static double rt_GetDblValueFromOverSizedData(
  316       const void *pVoid, 
  317       int bitsPerChunk, 
  318       int numOfChunk,
  319       unsigned int isSigned, 
  320       double fracSlope, 
  321       int fixedExp, 
  322       double bias)
  323   {
  324       double retValue = 0;
  325   
  326       double *dblValue = (double *) calloc(numOfChunk, sizeof(double));
  327   
  328       int i;    
  329       double isSignedNeg;
  330   
  331       if(isSigned) {
  332           const chunk_T *pData = (const chunk_T *) (pVoid);
  333           for (i = 0; i <numOfChunk; i++) {
  334               dblValue[i] = (double)(pData[i]);
  335           }
  336       } else  {
  337           const uchunk_T *pData = (const uchunk_T *) (pVoid);
  338           for (i = 0; i <numOfChunk; i++) {
  339               dblValue[i] = (double)(pData[i]);
  340           }
  341       }
  342   
  343       /* 
  344          Assuming multi chunks b_n ... b_2 b_1 b_0, and the length of each chunk is N.
  345          Suppose b_i is the i-th chunk's value.
  346          Then for unsigned data or data with one chunk: we have
  347          retValue = b_n * 2^(n*N) + ... + b_1 * 2^N + b_0 * 2^0;
  348          But for signed data, we have
  349          retValue = b_n * 2^(n*N) + ... + b_1 * 2^N + b_0 * 2^0+ (b_0<0) * 2^N + 
  350          ... (b_(n-1) <0) * 2^(n*N) 
  351          = (b_n + (b_(n-1)<0)) * 2^(n*N) +... + (b_1 + (b_0<0)) * 2^N + b_0 * 2^0;
  352          Together:
  353          retValue = 
  354          (b_n + isSigned * (b_(n-1)<0)) * 2^(n*N) +... + (b_1 + isSigned * (b_0<0)) * 2^N + b_0 * 2^0;
  355       */
  356   
  357       retValue = dblValue[numOfChunk - 1];
  358       
  359       for(i = numOfChunk - 1; i > 0; i--) {
  360           isSignedNeg = dblValue[i - 1] < 0 ? (double)isSigned : 0;
  361           retValue = retValue + isSignedNeg;
  362   
  363           retValue = ldexp(retValue, bitsPerChunk)+ dblValue[i-1];
  364       }
  365       retValue = ldexp( fracSlope * retValue, fixedExp ) + bias;
  366   
  367       FREE(dblValue);
  368       return (retValue);
  369   
  370   } /* end rt_GetDblValueFromOverSizedData */

  562   /* Function: rt_ProcessMatItem =================================================
  563    * Abstract:
  564    *      This routine along with rt_WriteItemToMatFile() write out a specified
  565    *      mat-item the .mat file. Note that if the input argument
  566    *          cmd == 0, then this function just calculates the size of the item.
  567    *          cmd <> 0, this function writes the mat-item to the file.
  568    *      Return values is
  569    *           -1 : coding/logic error
  570    *            0 : upon success
  571    *          > 0 : upon write failure (1)
  572    */
  573   static int_T rt_ProcessMatItem(FILE         *fp,
  574                                  MatItem      *pItem,
  575                                  ItemDataKind itemKind,
  576                                  int_T        cmd)
  577   {
  578       mxClassID    mxID          = mxUNKNOWN_CLASS;
  579       uint32_T     arrayFlags[2] = {0, 0};
  580       int32_T      *dims         = NULL;
  581       int32_T      _dims[3]      = {0, 0, 0};
  582       int32_T      nDims         = 2;
  583       int32_T      nBytesInItem  = 0;
  584       const char_T *itemName;
  585       MatItem      item;
  586       int_T        retStat       = 0;
  587   
  588       switch (itemKind) {
  589         case DATA_ITEM: {
  590             (void)fprintf(stderr,"Unexpected itemKind = DATA_ITEM in "
  591                                  "rt_ProcessMatItem @A\n");
  592             retStat = -1;
  593             goto EXIT_POINT;
  594         }
  595         case MATRIX_ITEM: {
  596             const MatrixData *var = (const MatrixData *) pItem->data;
  597   
  598             mxID           = var->mxID;
  599             arrayFlags[0]  = mxID;
  600             arrayFlags[0] |= var->logical;
  601             arrayFlags[0] |= var->complex;
  602             if (var->nDims < 2) {
  603                 dims         = _dims;
  604                 dims[0]      = var->nRows;
  605                 dims[1]      = var->nCols;
  606                 nDims        = 2;
  607             } else if (var->nDims >= 2) {
  608                 int32_T k;
  609                 dims = (int32_T*)malloc(sizeof(int32_T)*(var->nDims+1));
  610                 for (k = 0; k < var->nDims; k++) {
  611                     dims[k] = var->dims[k];
  612                 }
  613                 dims[var->nDims] = var->nRows;
  614                 nDims = var->nDims + 1;
  615             }
  616             itemName = var->name;
  617             break;
  618         }
  619         case STRUCT_LOG_VAR_ITEM: {
  620             const StructLogVar *var = (const StructLogVar *) pItem->data;
  621   
  622             mxID          = mxSTRUCT_CLASS;
  623             arrayFlags[0] = mxID;
  624             dims          = _dims;
  625             dims[0]       = 1;
  626             dims[1]       = 1;
  627             itemName      = var->name;
  628             break;
  629         }
  630         case SIGNALS_STRUCT_ITEM: {
  631             const SignalsStruct *var = (const SignalsStruct *) pItem->data;
  632   
  633             mxID          = mxSTRUCT_CLASS;
  634             arrayFlags[0] = mxID;
  635             dims          = _dims;
  636             dims[0]       = 1;
  637             dims[1]       = var->numSignals;
  638             itemName      = &SIGNALS_FIELD_NAME;
  639             break;
  640         }
  641         default:
  642           (void)fprintf(stderr,"Unexpected itemKind=%d in rt_ProcessMatItem @B\n",
  643                         itemKind);
  644           retStat = -1;
  645           goto EXIT_POINT;
  646       }
  647   
  648       /* array flags */
  649       item.nbytes = 2*sizeof(uint32_T);
  650       if (cmd) {
  651           item.type = matUINT32;
  652           item.data = arrayFlags;
  653           if (rt_WriteItemToMatFile(fp,&item, DATA_ITEM)) {
  654               retStat = 1;
  655               goto EXIT_POINT;
  656           }
  657       } else {
  658           /*LINTED E_CAST_INT_TO_SMALL_INT*/
  659           nBytesInItem += matINT64_ALIGN(matTAG_SIZE + item.nbytes);
  660       }
  661       /* dimensions */
  662       /*LINTED E_ASSIGN_INT_TO_SMALL_INT*/
  663       item.nbytes = nDims*sizeof(int32_T);
  664       if (cmd) {
  665           item.type = matINT32;
  666           item.data = dims;
  667           if (rt_WriteItemToMatFile(fp,&item, DATA_ITEM)) {
  668               retStat = 1;
  669               goto EXIT_POINT;            
  670           }
  671       } else {
  672           /*LINTED E_CAST_INT_TO_SMALL_INT*/
  673           nBytesInItem += matINT64_ALIGN(matTAG_SIZE + item.nbytes);
  674       }
  675       /* name */
  676       item.nbytes = (int32_T)strlen(itemName);
  677       if (cmd) {
  678           item.type = matINT8;
  679           item.data = (const char_T*) itemName;
  680           if (rt_WriteItemToMatFile(fp,&item, DATA_ITEM)) {
  681               retStat = 1;
  682               goto EXIT_POINT;
  683           }
  684       } else {
  685           nBytesInItem += (item.nbytes <= 4) ? /*LINTED E_CAST_INT_TO_SMALL_INT*/
  686                           matTAG_SIZE : matINT64_ALIGN(matTAG_SIZE + item.nbytes);
  687       }
  688   
  689       if (itemKind == MATRIX_ITEM) {
  690           const MatrixData *var   = (const MatrixData*) pItem->data;
  691           int_T            matID  = rt_GetMatIdFromMxId(mxID);
  692           size_t           elSize = var->elSize;
  693   
  694           /* data */
  695           item.nbytes = (int32_T)(var->nRows * var->nCols * elSize);
  696           if (cmd) {
  697               item.type = matID;
  698               item.data = var->re;
  699               if (rt_WriteItemToMatFile(fp, &item, DATA_ITEM)) {
  700                   retStat = 1;
  701                   goto EXIT_POINT;
  702               }
  703           } else {
  704               nBytesInItem += (item.nbytes <= 4) ? /*LINTED*/
  705                           matTAG_SIZE : matINT64_ALIGN(matTAG_SIZE + item.nbytes);
  706           }
  707           /* imaginary part */
  708           if (var->complex) {
  709               item.nbytes = (int32_T)(var->nRows * var->nCols * elSize);
  710               if (cmd) {
  711                   item.type = matID;
  712                   item.data = var->im;
  713                   if (rt_WriteItemToMatFile(fp, &item, DATA_ITEM)) {
  714                       retStat = 1;
  715                       goto EXIT_POINT;
  716                   }
  717               } else {
  718                   nBytesInItem += (item.nbytes <= 4) ? /*LINTED*/
  719                           matTAG_SIZE : matINT64_ALIGN(matTAG_SIZE + item.nbytes);
  720               }
  721           }

  724           int_T        sizeofFieldNames;

  726           /* field names */

  744               goto EXIT_POINT;
  745           }
  746   
  747           /* write field names */
  748           if (cmd) {
  749               int32_T tmpInt = mxMAXNAM;
  750   
  751               item.nbytes = sizeof(int32_T);
  752               item.type   = matINT32;
  753               item.data   = &tmpInt;
  754               if (rt_WriteItemToMatFile(fp,&item, DATA_ITEM)) {
  755                   retStat = 1;
  756                   goto EXIT_POINT;
  757               }
  758   
  759               item.nbytes = sizeofFieldNames;
  760               item.type   = matINT8;
  761               item.data   = (const char_T*) fieldNames;

  763                   retStat = 1;
  764                   goto EXIT_POINT;
  765               }
  766           } else {
  767               /*LINTED E_CAST_INT_TO_SMALL_INT*/

  769                                               sizeofFieldNames );

  772           /* process each field of the structure */

  774             case STRUCT_LOG_VAR_ITEM: {

 1050   /* Function: rt_WriteItemToMatFile =============================================
 1051    * Abstract:
 1052    *      Entry function for writing out a mat item to the mat file.
 1053    *
 1054    *      Return values is
 1055    *          == 0 : upon success
 1056    *          <> 0 : upon failure
 1057    */
 1058   static int_T rt_WriteItemToMatFile(FILE         *fp,
 1059                                      MatItem      *pItem,
 1060                                      ItemDataKind itemKind)
 1061   {
 1062       /* Determine the item size */
 1063       if (pItem->type == matMATRIX) {
 1064           if (rt_ProcessMatItem(fp, pItem, itemKind, 0)) return(1);
 1065       }
 1066   
 1067       /* Write the item tag and data */
 1068       if (pItem->nbytes > 4) {
 1069           int32_T nAlignBytes;
 1070   
 1071           if (fwrite(pItem, 1, matTAG_SIZE, fp) != matTAG_SIZE) return(1);
 1072   
 1073           if (pItem->type == matMATRIX) {
 1074               if (rt_ProcessMatItem(fp, pItem, itemKind, 1)) return(1);
 1075           } else {
 1076               if ( fwrite(pItem->data, 1, pItem->nbytes, fp) !=
 1077                                                       ((size_t) pItem->nbytes) ) {
 1078                   return(1);
 1079               }
 1080           }
 1081   
 1082           /* Add offset for 8-byte alignment */
 1083           nAlignBytes = matINT64_ALIGN(pItem->nbytes) - pItem->nbytes;
 1084           if (nAlignBytes > 0) {
 1085               int pad[2] = {0, 0};
 1086               if ( fwrite(pad,1,nAlignBytes,fp) != ((size_t) nAlignBytes) ) {
 1087                   return(1);
 1088               }
 1089           }
 1090       } else {
 1091           MatItem item = {0, 0, NULL};
 1092           item.type = ((uint32_T)(pItem->type))|(((uint32_T)(pItem->nbytes))<<16);
 1093           (void)memcpy(&item.nbytes, pItem->data, pItem->nbytes);
 1094           if (fwrite(&item, 1, matTAG_SIZE, fp) != matTAG_SIZE) return(1);
 1095       }
 1096   
 1097       return(0);
 1098   
 1099   } /* end rt_WriteItemToMatFile */

 1102   /* Function: rt_WriteMat5FileHeader ============================================
 1103    * Abstract:
 1104    *      Function to write the mat file header.
 1105    *      Return values is
 1106    *          == 0 : upon success
 1107    *          <> 0 : upon failure
 1108    */
 1109   static int_T rt_WriteMat5FileHeader(FILE *fp)
 1110   {
 1111       int_T        nbytes;
 1112       int_T        nspaces;
 1113       int_T        i, n;
 1114       unsigned short ver[2];
 1115       char_T       spaces[16];
 1116       const char_T *matversion = "MATLAB 5.0 MAT-file";
 1117   
 1118       (void)memset(spaces, ' ', sizeof(spaces));
 1119   
 1120       n = (int_T)strlen(matversion);
 1121       nbytes = (int_T)fwrite(matversion, 1, n, fp);
 1122       /*LINTED E_ASSIGN_INT_TO_SMALL_INT*/
 1123       nspaces = matVERSION_INFO_OFFSET - nbytes;
 1124       /*LINTED E_ASSIGN_INT_TO_SMALL_INT*/
 1125       n = nspaces % sizeof(spaces);
 1126       nbytes += (int_T)fwrite(spaces, 1, n, fp);
 1127       /*LINTED E_ASSIGN_INT_TO_SMALL_INT*/
 1128       n = nspaces / sizeof(spaces);
 1129       for (i = 0; i < n; ++i) {
 1130           nbytes += (int_T)fwrite(spaces, 1, sizeof(spaces), fp);
 1131       }
 1132       if (nbytes == matVERSION_INFO_OFFSET) {
 1133           ver[0] = matVERSION;
 1134           ver[1] = matKEY;
 1135           nbytes += (int_T)fwrite(ver, 1, sizeof(ver), fp);
 1136       }
 1137       return(nbytes != matVERSION_INFO_OFFSET + sizeof(ver));
 1138   
 1139   } /* end rt_WriteMat5FileHeader */

 1149       int_T  maxRows = var->data.nRows;
 1150       int_T  nDims   = var->data.nDims;
 1151       size_t elSize  = var->data.elSize;
 1152       int_T  nRows   = (var->wrapped ?  maxRows : var->rowIdx);
 1153   
 1154       var->nDataPoints = var->rowIdx + var->wrapped * maxRows;
 1155   
 1156       if (var->wrapped > 1 || (var->wrapped == 1 && var->rowIdx != 0)) {
 1157           /*
 1158            * Warn the user the circular buffer has wrapped, implying that
 1159            * some data has been lost.
 1160            */
 1161           if( verbose) {
 1162               (void)fprintf(stdout,
 1163                             "*** Log variable %s has wrapped %d times\n"
 1164                             "    using a circular buffer of size %d\n",
 1165                             var->data.name, var->wrapped, var->data.nRows);
 1166           }
 1167           if (var->usingDefaultBufSize) {
 1168               /*
 1169                * If wrapping occurred using the default buffer size,
 1170                * let the user know what size buffer to use in the
 1171                * future to avoid wrapping.  If the default buffer
 1172                * size was not used, the user has no control to specify
 1173                * the correct value.  Wrapping may occur when not using
 1174                * the default buffer if we allocated too small a buffer
 1175                * size for this logvar.  One common case is a toWorkspace
 1176                * block inside of an iterative subsystem - we can not take
 1177                * the number of iterations into account (they may be
 1178                * variable) when allocating the buffer.  In this case,
 1179                * just warn the buffer wrapped and don't tell user they
 1180                * can override the buffer size.
 1181                */
 1182               if( verbose ) {
 1183                   (void)fprintf(stdout,
 1184                                 "*** To avoid wrapping, explicitly specify a\n"
 1185                                 "    buffer size of %d in your Simulink model\n"
 1186                                 "    by adding OPTS=\"-DDEFAULT_BUFFER_SIZE=%d\"\n"
 1187                                 "    as an argument to the ConfigSet MakeCommand\n"
 1188                                 "    parameter\n",
 1189                                 var->nDataPoints, var->nDataPoints);
 1190               }
 1191           }
 1192       }
 1193   
 1194       if (nDims < 2 && nCols > 1) {  /* Transpose? */
 1195           /* Don't need to transpose valueDimensions */
 1196           int_T  nEl    = nRows*nCols;
 1197           char   *src   = var->data.re;
 1198           char   *pmT;
 1199           int_T  k;
 1200   
 1201           /**********************************
 1202            * If memory cannot be allocated, *
 1203            * write to a temporary buffer    *
 1204            **********************************/
 1205           if ((pmT = malloc(nEl*elSize)) == NULL) {
 1206               FILE  *fptr;
 1207               char  fName[mxMAXNAM];
 1208   
 1209               (void)sprintf(fName, "%s%s", var->data.name, "_rtw_tmw.tmw");
 1210               if ((fptr=fopen(fName,"w+b")) == NULL) {
 1211                   (void)fprintf(stderr,"*** Error opening %s",fName);
 1212                   return("unable to open data file\n");
 1213               }
 1214   
 1215               /****************************
 1216                * Write the data to a file *
 1217                ****************************/
 1218               for (k=0; k<nEl; k++) {
 1219                   int_T kT   = nCols*(k%nRows) + (k/nRows);
 1220                   char  *dst = src + kT*elSize;
 1221                   (void)fwrite(dst, elSize, 1, fptr);
 1222               }
 1223               if (var->data.complex) {
 1224                   char *pmiT = var->data.re;
 1225                   src  = var->data.im;
 1226                   for (k=0; k<nEl; k++) {
 1227                       int_T kT   = nRows*(k%nCols) + (k/nCols);
 1228                       char  *dst = pmiT + kT*elSize;
 1229                       (void)memcpy(dst, src, elSize);
 1230                       src += elSize;
 1231                   }
 1232                   var->data.re = var->data.im;
 1233                   var->data.im = pmiT;
 1234               }
 1235   
 1236               /*******************************
 1237                * Read the data from the file *
 1238                *******************************/
 1239               (void)rewind(fptr);
 1240               (void)fread(var->data.re, elSize, nEl, fptr);
 1241               (void)fclose(fptr);
 1242               (void)remove(fName);
 1243           } else {
 1244               for (k=0; k<nEl; k++) {
 1245                   int_T kT   = nRows*(k%nCols) + (k/nCols);
 1246                   char  *dst = pmT + kT*elSize;
 1247                   (void)memcpy(dst, src, elSize);
 1248                   src += elSize;
 1249               }
 1250               if (var->data.complex) {
 1251                   char *pmiT = var->data.re;
 1252                   src  = var->data.im;
 1253                   for (k=0; k<nEl; k++) {
 1254                       int_T kT   = nRows*(k%nCols) + (k/nCols);
 1255                       char  *dst = pmiT + kT*elSize;
 1256                       (void)memcpy(dst, src, elSize);
 1257                       src += elSize;
 1258                   }
 1259                   var->data.re = var->data.im;
 1260                   var->data.im = pmiT;
 1261               }
 1262               FREE(var->data.re);
 1263               var->data.re = pmT;
 1264           }
 1265       } /* Transpose? */
 1266   
 1267       if (var->wrapped > 0 && var->rowIdx != 0 ) {  /* Rotate? */
 1268           char_T *buffer    = var->data.re;
 1269           int_T  done       = 0; /* done: 0 (1) rotate real (imag) part. */
 1270   
 1271           do {
 1272               char_T *col       = buffer;
 1273               int_T  rowOffset  = (int_T)((nDims == 1) ? (elSize) : (elSize * nCols));
 1274               int_T  colOffset  = (int_T)((nDims == 1)?  (nRows*elSize) : elSize);
 1275               int_T  zeroIdx    = var->rowIdx;
 1276               int_T  j;
 1277   
 1278               for (j = 0 ; j < nCols; ++j, col += colOffset) {
 1279                   int_T   swapCount;
 1280                   int_T   srcIdx;
 1281                   int_T   dstIdx;
 1282                   int_T   tmpIdx;
 1283                   MatReal tmp;
 1284   
 1285                   for (tmpIdx=0, swapCount=0; swapCount < nRows; tmpIdx++) {
 1286                       (void)memcpy(&tmp, col + tmpIdx*rowOffset, elSize);
 1287   
 1288                       dstIdx=tmpIdx; 
 1289                       srcIdx = ((dstIdx + zeroIdx) % nRows);
 1290                       while (srcIdx != tmpIdx) {
 1291                           (void)memcpy(col + dstIdx*rowOffset,
 1292                                        col + srcIdx*rowOffset,
 1293                                        elSize);
 1294                           ++swapCount;
 1295                           dstIdx = srcIdx;
 1296                           srcIdx = ((dstIdx + zeroIdx) % nRows);
 1297                           
 1298                       }
 1299                       (void)memcpy(col + dstIdx*rowOffset, &tmp, elSize);
 1300                       ++swapCount;
 1301                   }
 1302               }
 1303               done ++;
 1304               /* need to rotate the imaginary part */
 1305           } while ((done == 1) && ((buffer = var->data.im) != NULL));
 1306   
 1307           var->rowIdx = 0;
 1308       } /* Rotate? */
 1309   
 1310       /*
 1311        * We might have allocated more number of rows than the number of data
 1312        * points that have been logged, in which case set nRows to nDataPoints
 1313        * so that only these values get saved.
 1314        */
 1315       if (var->nDataPoints < var->data.nRows) {
 1316           var->data.nRows = var->nDataPoints;
 1317           if(var->valDims != NULL){
 1318               size_t elSizeValDims = sizeof(real_T);
 1319               int_T  k;
 1320               real_T *dimsData = var->valDims->dimsData + nRows;
 1321               /* 
 1322                  Keep nRows of values and that of valueDimensions consistent 
 1323                  for variable-size signals.
 1324               */
 1325               var->valDims->nRows = var->data.nRows;
 1326               /*
 1327                  Also need to move data when shrinking the array size,
 1328                  because valueDimensions data is stored in array format. 
 1329                  e.g. maxRows = 4; nRows = 2; nDims = 3;
 1330                  Before fixing up the logVar, the locations of data are as below:
 1331                  (x, y, z -- useful data / o -- junk)
 1332                  a[0] = x    a[4] = y    a[8] = z
 1333                  a[1] = x    a[5] = y    a[9] = z
 1334                  a[2] = o    a[6] = o    a[10]= o
 1335                  a[3] = o    a[7] = o    a[11]= o
 1336                  After fixing up the logVar, we want the data to be stored as:
 1337                  a[0] = x    a[4] = z    a[8] = o
 1338                  a[1] = x    a[5] = z    a[9] = o
 1339                  a[2] = y    a[6] = o    a[10]= o
 1340                  a[3] = y    a[7] = o    a[11]= o
 1341               */
 1342               for(k = 1; k < nDims; k++){
 1343                   (void) memcpy(dimsData, 
 1344                                 var->valDims->dimsData + k*maxRows,
 1345                                 elSizeValDims * nRows);
 1346                   dimsData += nRows;
 1347               }
 1348           }
 1349       }
 1350       return(NULL);
 1351   
 1352   } /* end rt_FixupLogVar */

 1355   /* Function: rt_LoadModifiedLogVarName =========================================
 1356    * Abstract:
 1357    *      The name of the logged variable is obtained from the input argument
 1358    *      varName and the nameModifier which is obtained from the simstruct. If
 1359    *      the nameModifier begins with an '_', then nameModifier is post-pended to
 1360    *      varName to obtain the name of the logged variable. If the first
 1361    *      character does not begin with an '_', then the nameModifier is
 1362    *      pre-pended to varName.
 1363    *
 1364    * Examples:
 1365    *     a)  varName = "tout" & nameModifier = "_rt"  => logVarName = "tout_rt"
 1366    *     b)  varName = "tout" & nameModifier = "rt_"  => logVarName = "rt_tout"
 1367    *     c)  varName = "tout" & nameModifier = "none" => logVarName = "tout"
 1368    */
 1369   static void rt_LoadModifiedLogVarName(const RTWLogInfo *li,         /* in  */
 1370                                         const char       *varName,    /* in  */
 1371                                         char             *logVarName) /* out */
 1372   {
 1373       int_T        nameLen;
 1374       const char_T *nameModifier = rtliGetLogVarNameModifier(li);
 1375   
 1376       if (nameModifier != NULL && strcmp(nameModifier,"none")==0) {
 1377           nameModifier = NULL;
 1378       }
 1379   
 1380       logVarName[mxMAXNAM-1] = '\0';
 1381       if (nameModifier == NULL) {
 1382           (void)strncpy(logVarName, varName, mxMAXNAM-1);
 1383       } else if (nameModifier[0] == '_') {
 1384           (void)strncpy(logVarName, varName, mxMAXNAM-1);
 1385           nameLen = (int_T)strlen(logVarName);
 1386           (void)strncat(logVarName, nameModifier, mxMAXNAM-1-nameLen);
 1387       } else {
 1388           (void)strncpy(logVarName, nameModifier, mxMAXNAM-1);
 1389           nameLen = (int_T)strlen(logVarName);
 1390           (void)strncat(logVarName, varName, mxMAXNAM-1-nameLen);
 1391       }
 1392   
 1393   } /* end rt_LoadModifiedLogVarName */

 1697               mtxData->logical = 0U;
 1698               mtxData->complex = 0U;
 1699

 1703       }

 1706       if (plotStyles != NULL) {
 1707           real_T         *data;
 1708           size_t	       nbytes;
 1709           int_T          dataLen    = 0;
 1710           BuiltInDTypeId dTypeId    = rt_GetActualDTypeID(SS_DOUBLE);
 1711           /*LINTED E_ASSIGN_INT_TO_SMALL_INT*/

 2360                         (double)nRows*nCols*elSize,
 2361                         (double)(nRows+DEFAULT_BUFFER_SIZE)*nCols*elSize);

 2363       }

 2461   /* Function: rt_UpdateLogVarWithDiscontiguousData ==============================
 2462    * Abstract:
 2463    *      Log one row of the LogVar with data that is not contiguous.
 2464    */
 2465   static void rt_UpdateLogVarWithDiscontiguousData(LogVar            *var,
 2466                                                    LogSignalPtrsType data,
 2467                                                    const int_T       *segmentLengths,
 2468                                                    int_T             nSegments)
 2469   {
 2470       size_t elSize = 0;
 2471       int_T  offset = 0;
 2472       int    segIdx = 0;
 2473   
 2474       if (++var->numHits % var->decimation) return;
 2475       var->numHits = 0;
 2476   
 2477       /*
 2478        * Reallocate or wrap the LogVar
 2479        */
 2480       if (var->rowIdx == var->data.nRows) {
 2481           if (var->okayToRealloc == 1) {
 2482               rt_ReallocLogVar(var, false);
 2483           } else {
 2484               /* Circular buffer */
 2485               var->rowIdx = 0;
 2486               ++(var->wrapped); /* increment the wrap around counter */
 2487           }
 2488       }
 2489   
 2490       /* This function is only used to log states, there's no var-dims issue. */
 2491       elSize = var->data.elSize;
 2492       offset = (int_T)(elSize * var->rowIdx * var->data.nCols);
 2493   
 2494       if (var->data.complex) {
 2495           char_T *dstRe = (char_T*)(var->data.re) + offset;
 2496           char_T *dstIm = (char_T*)(var->data.im) + offset;
 2497   
 2498           for (segIdx = 0; segIdx < nSegments; segIdx++) {
 2499               int_T         nEl  = segmentLengths[segIdx];
 2500               const  char_T *src = (const void *)data[segIdx];
 2501               int_T         el;
 2502   
 2503               for (el = 0; el < nEl; el++) {
 2504                   (void)memcpy(dstRe, src, elSize);
 2505                   dstRe += elSize;   src += elSize;
 2506                   (void)memcpy(dstIm, src, elSize);
 2507                   dstIm += elSize;   src += elSize;
 2508               }
 2509           }
 2510       } else {
 2511           char_T *dst = (char_T*)(var->data.re) + offset;
 2512   
 2513           for (segIdx = 0; segIdx < nSegments; segIdx++) {
 2514               size_t      segSize = elSize*segmentLengths[segIdx];
 2515               const  void *src    = data[segIdx];
 2516   
 2517               (void)memcpy(dst, src, segSize);
 2518               dst += segSize;
 2519           }
 2520       }
 2521   
 2522       ++var->rowIdx;
 2523       return;
 2524   
 2525   } /* end rt_UpdateLogVarWithDiscontiguousData */

 2539   /* Function: rt_CreateLogVarWithConvert ========================================
 2540    * Abstract:
 2541    *	Create a logging variable.
 2542    *
 2543    * Returns:
 2544    *	~= NULL  => success, returns the log variable created.
 2545    *	== NULL  => failure, error message set in the simstruct.
 2546    */
 2547   LogVar *rt_CreateLogVarWithConvert(
 2548       RTWLogInfo        *li,
 2549       const real_T      startTime,
 2550       const real_T      finalTime,
 2551       const real_T      inStepSize,
 2552       const char_T      **errStatus,
 2553       const char_T      *varName,
 2554       BuiltInDTypeId    inpDataTypeID,
 2555       const RTWLogDataTypeConvert *pDataTypeConvertInfo,
 2556       int_T             logical,
 2557       int_T             complex,
 2558       int_T             frameData,
 2559       int_T             nCols,
 2560       int_T             nDims,
 2561       const int_T       *dims,
 2562       LogValDimsStat    logValDimsStat,
 2563       void              **currSigDims,
 2564       int_T             *currSigDimsSize,
 2565       int_T             maxRows,
 2566       int_T             decimation,
 2567       real_T            sampleTime,
 2568       int_T             appendToLogVarsList)
 2569   {
 2570       int_T          usingDefaultBufSize = 0;
 2571   #ifdef NO_LOGGING_REALLOC
 2572       int_T          okayToRealloc       = 0;
 2573   #else
 2574       int_T          okayToRealloc       = 1;
 2575   #endif
 2576       LogVar         *var                = NULL;
 2577       /*inpDataTypeID is the rt_LoggedOutputDataTypeId*/
 2578       BuiltInDTypeId dTypeID             = (BuiltInDTypeId)inpDataTypeID; 
 2579       size_t         elementSize         = rt_GetSizeofDataType(dTypeID);
 2580       int_T          frameSize;
 2581       int_T          nRows;
 2582       int_T          nColumns;
 2583   
 2584       /*===================================================================*
 2585        * Determine the frame size if the data is frame based               *
 2586        *===================================================================*/
 2587       frameSize = frameData ? dims[0] : 1;
 2588   
 2589       /*===================================================================*
 2590        * Calculate maximum number of rows needed in the buffer             *
 2591        *===================================================================*/
 2592   
 2593       if (finalTime > startTime && finalTime != rtInf) {
 2594           real_T nPoints;            /* Tfinal is finite  ===>  nRows can be  */
 2595           real_T stepSize;           /* computed since the StepSize is fixed  */
 2596   
 2597           if (sampleTime == -2.0) {  /* The signal being logged is constant,  *
 2598                                       * Hence, only one data point is logged. */
 2599               stepSize = finalTime;
 2600           } else if (sampleTime == -1.0 || sampleTime == 0.0) {
 2601                                   /* Signal being logged is either inside a    *
 2602                                    * triggered sub-system or it is continuous. */
 2603               stepSize = inStepSize;
 2604           } else {                /* Discrete signal */
 2605               stepSize = sampleTime;
 2606           }
 2607   
 2608           if (stepSize == 0.0) {
 2609               /* small initial value, so as to exercise the realloc code */
 2610               nRows = maxRows+1;
 2611               okayToRealloc = 1;
 2612           } else {
 2613               nPoints = 1.0 + floor((finalTime-startTime)/stepSize);
 2614   
 2615               /*
 2616                * Add one more data point if needed.
 2617                */
 2618               if ( stepSize*(nPoints-1.0) < (finalTime-startTime) ) {
 2619                   nPoints += 1.0;
 2620               }
 2621   
 2622               /*
 2623                * Actual number of points to log = nPoints * size of
 2624                * each frame if data is frame-based
 2625                */
 2626               nPoints = frameData ? (nPoints * frameSize) : nPoints;
 2627   
 2628               nPoints /= decimation;
 2629               if (nPoints != floor(nPoints)) {
 2630                   nPoints += 1.0;
 2631               }
 2632               nRows = (nPoints <= INT_MAX) ? ((int_T) nPoints) : INT_MAX;
 2633           }
 2634           /*
 2635            * If maxRows is specified, and if this number is less
 2636            * than the number we computed (nRows) then use maxRows.
 2637            */
 2638           if ((maxRows > 0) && (maxRows < nRows)) {
 2639               nRows = maxRows;
 2640               okayToRealloc = 0;
 2641           }
 2642       } else if (finalTime == startTime) {
 2643           /*
 2644            * Number of rows to log is equal to 1 if not frame-based and
 2645            * equal to frame size if frame-based
 2646            */
 2647           nRows = frameData ? frameSize : 1;
 2648   
 2649           /*
 2650            * If maxRows is specified, and if this number is less
 2651            * than the number we computed (nRows) then use maxRows.
 2652            */
 2653           if ((maxRows > 0) && (maxRows < nRows)) {
 2654               nRows = maxRows;
 2655               okayToRealloc = 0;
 2656           }
 2657       } else if (maxRows > 0) {     /* maxRows is specified => nRows=maxRows  */
 2658           nRows = maxRows;
 2659           okayToRealloc = 0;
 2660       } else {
 2661   
 2662           if (inStepSize == 0) {
 2663               /* small initial value, so as to exercise the realloc code */
 2664               nRows = maxRows+1;
 2665               okayToRealloc = 1;
 2666           } else {                    /* Use a default value for nRows          */
 2667               usingDefaultBufSize = 1;
 2668               nRows = DEFAULT_BUFFER_SIZE;
 2669               okayToRealloc = 0;  /* No realloc with infinite stop time */
 2670               (void)fprintf(stdout, "*** Using a default buffer of size %d for "
 2671                             "logging variable %s\n", nRows, varName);
 2672           }
 2673       }
 2674   
 2675       /*
 2676        * Figure out the number of columns that the log variable should have.
 2677        * If the data is not frame based, then number of columns should equal
 2678        * nCols that is provided as input to the function. If the data is
 2679        * frame-based, then the number of columns should be equal to the
 2680        * number of channels = nCols/frameSize = dims[1];
 2681        */
 2682       nColumns = frameData ? dims[1] : nCols;
 2683   
 2684       /*
 2685        * Error out if the size of the circular buffer is absurdly large, this
 2686        * error message is more informative than the one we get when we try to
 2687        * malloc this many number of bytes in one fell swoop.
 2688        */
 2689       {
 2690           double tmpDbl = ((double)elementSize)*((double)nRows)*
 2691                                                 ((double)nColumns);
 2692   
 2693           if (tmpDbl >= UINT_MAX) {
 2694               (void)fprintf(stderr,
 2695                             "\n*** Memory required to log variable '%s' is too"
 2696                             "\n    big. Use the 'Limit rows to last:' and (or)"
 2697                             "\n    'Decimation:' options to reduce the required"
 2698                             "\n    memory size.\n", varName);
 2699               (void)fprintf(stderr, "*** Details:\n"
 2700                             "       varName         = %s\n"
 2701                             "       nRows           = %d\n"
 2702                             "       nCols           = %d\n"
 2703                             "       elementSize     = %lu\n"
 2704                             "       Bytes Required  = %.16g\n\n",
 2705                             varName, nRows, nColumns, (unsigned long)
 2706                             elementSize, tmpDbl);
 2707               goto ERROR_EXIT;
 2708           }
 2709       }
 2710   
 2711       /* Allocate memory for the log variable */
 2712       if ( (var = calloc(1, sizeof(LogVar))) == NULL ) {
 2713           (void)fprintf(stderr, "*** Error allocating memory for logging %s\n",
 2714                         varName);
 2715           goto ERROR_EXIT;
 2716       }
 2717   
 2718       /* Allocate memory for the circular buffer (real part) */
 2719       if ( (var->data.re = malloc(nRows*nColumns*elementSize)) == NULL ) {
 2720           (void)fprintf(stderr,
 2721                         "*** Error allocating memory for the circular buffer\n");
 2722           (void)fprintf(stderr, "*** Details:\n"
 2723                         "       varName         = %s\n"
 2724                         "       nRows           = %d\n"
 2725                         "       nCols           = %d\n"
 2726                         "       elementSize     = %lu\n"
 2727                         "       Bytes Requested = %.16g\n\n",
 2728                         varName, nRows, nColumns, (unsigned long) elementSize,
 2729                         ((double)elementSize)*((double)nRows)*((double)nColumns));
 2730           goto ERROR_EXIT;
 2731       }
 2732   
 2733       /* Allocate memory for the circular buffer for the imaginary part */
 2734       if (complex) {
 2735           if ( (var->data.im = malloc(nRows*nColumns*elementSize)) == NULL ) {
 2736               (void)fprintf(stderr,
 2737                             "*** Error allocating memory for the circular buffer "
 2738                             "for logging the imaginary part of %s\n", varName);
 2739               (void)fprintf(stderr, "*** Details:\n"
 2740                             "       varName         = %s\n"
 2741                             "       nRows           = %d\n"
 2742                             "       nCols           = %d\n"
 2743                             "       elementSize     = %lu\n"
 2744                             "       Bytes Requested = %.16g\n\n",
 2745                             varName, nRows, nColumns, (unsigned long) elementSize,
 2746                             ((double)elementSize)*((double)nRows)*
 2747                                                    ((double)nColumns));
 2748               goto ERROR_EXIT;
 2749           }
 2750       }
 2751       /*
 2752        * Initialize the fields in LogVar structure.
 2753        */
 2754       if (appendToLogVarsList) {
 2755           rt_LoadModifiedLogVarName(li,varName,var->data.name);
 2756       } else {

 2758           (void)strncpy(var->data.name,varName,mxMAXNAM-1);
 2759       }
 2760       var->data.nCols           = nColumns;
 2761       var->data.nRows           = nRows;
 2762   
 2763       var->data.nDims           = frameData ? 1 : nDims;
 2764       if (var->data.nDims > 2) {
 2765           var->data.dims = (int_T*)malloc(sizeof(int_T)*var->data.nDims);
 2766       } else {
 2767           var->data.dims = var->data._dims;
 2768       }
 2769       if (frameData) {
 2770           var->data.dims[0] = nColumns;
 2771       } else {
 2772           /*LINTED E_CAST_INT_TO_SMALL_INT*/
 2773           (void)memcpy(var->data.dims, dims, (size_t)(nDims*sizeof(int_T)));
 2774       }
 2775   
 2776       var->data.dTypeID         = dTypeID;
 2777       var->data.elSize          = elementSize;
 2778   
 2779       var->data.dataTypeConvertInfo = rt_GetDataTypeConvertInfo(
 2780           pDataTypeConvertInfo, dTypeID);
 2781   
 2782       var->data.mxID            = rt_GetMxIdFromDTypeId(dTypeID);
 2783       /* over-ride logical bit if data type is boolean */
 2784       logical = dTypeID == SS_BOOLEAN ? 1 : 0;
 2785       var->data.logical         = (logical)   ? matLOGICAL_BIT : 0x0;
 2786       var->data.complex         = (complex)   ? matCOMPLEX_BIT : 0x0;
 2787       var->data.frameData       = frameData;
 2788       var->data.frameSize       = (frameData) ? frameSize : 1;
 2789   
 2790       /* fill up valDims field */
 2791       if(logValDimsStat == NO_LOGVALDIMS){
 2792           /* All signals are fixed-size, no need to log valueDimensions field */
 2793           var->valDims     = NULL;
 2794           /* Set these pointers to NULLs in this case */
 2795           var->coords      = NULL;
 2796           var->strides     = NULL;
 2797           var->currStrides = NULL;
 2798       }
 2799       else{
 2800           if ( (var->valDims = calloc(1, sizeof(ValDimsData))) == NULL ) {
 2801               goto ERROR_EXIT;
 2802           }
 2803   
 2804           (void)memcpy(var->valDims->name, &VALUEDIMENSIONS_FIELD_NAME, mxMAXNAM);
 2805   
 2806           if (logValDimsStat == LOGVALDIMS_EMPTYMX) {
 2807               /* At least one signal is variable-size, 
 2808                  but the current signal is fixed-size. 
 2809                  Therefore, create a dummy MatrixData to write out valueDimensions 
 2810                  as an empty matrix. 
 2811               */
 2812               var->valDims->nRows = 0;
 2813               var->valDims->nCols = 0;
 2814               var->valDims->currSigDims = NULL;
 2815               var->valDims->currSigDimsSize = NULL;
 2816               var->valDims->dimsData = NULL;
 2817               /* Set these pointers to NULLs in this case */
 2818               var->coords      = NULL;
 2819               var->strides     = NULL;
 2820               var->currStrides = NULL;
 2821           } else { /* The current signal is a variable-size signal. */
 2822               /* The "valueDimensions" must be double, so re-assign element size */
 2823               elementSize = sizeof(real_T);
 2824   
 2825               /* When signals are frame-based, 'valueDimensions' has 1 column */
 2826               if(frameData){
 2827                   /* When signal is frame-based, the first dimension is always fixed, 
 2828                      so we only need to record the second dimension.
 2829                      e.g. Two frame-based signals [10x4] and [10x3], 
 2830                      'valueDimensions' and 'currSigDims'
 2831                      only record 4 or 3.
 2832                   */
 2833                   nColumns = 1;
 2834                   var->valDims->currSigDims = (void**) (currSigDims + 1);
 2835                   var->valDims->currSigDimsSize = (int_T*) (currSigDimsSize + 1);
 2836               } else { /* non-frame based */
 2837                   nColumns = nDims;
 2838                   var->valDims->currSigDims = (void**) currSigDims;
 2839                   var->valDims->currSigDimsSize = (int_T*) currSigDimsSize;
 2840               }
 2841               
 2842               /* Allocate memory for the circular buffer */
 2843               if ( (var->valDims->dimsData = malloc(nRows*nColumns*elementSize)) == NULL ) {
 2844                   (void)fprintf(stderr,
 2845                                 "*** Error allocating memory for the circular buffer\n");
 2846                   (void)fprintf(stderr, "*** Details:\n"
 2847                                 "       varName         = %s\n"
 2848                                 "       nRows           = %d\n"
 2849                                 "       nCols           = %d\n"
 2850                                 "       elementSize     = %lu\n"
 2851                                 "       Bytes Requested = %.16g\n\n",
 2852                                 var->valDims->name, nRows, nColumns, (unsigned long) elementSize,
 2853                                 ((double)elementSize)*((double)nRows)*((double)nColumns));
 2854                   goto ERROR_EXIT;
 2855               }
 2856               var->valDims->nRows = nRows;
 2857               var->valDims->nCols = nColumns;
 2858

 2860               {
 2861                   size_t nbytes = var->data.nDims*sizeof(int_T);
 2862                   if( ((var->coords = calloc(nbytes, 1)) == NULL)
 2863                       ||((var->strides = calloc(nbytes, 1)) == NULL)
 2864                       ||((var->currStrides = calloc(nbytes, 1)) == NULL) )
 2865                       goto ERROR_EXIT;
 2866               }
 2867           }
 2868       }
 2869   
 2870       var->rowIdx               = 0;
 2871       var->wrapped              = 0;
 2872       var->nDataPoints          = 0;
 2873       var->usingDefaultBufSize  = usingDefaultBufSize;
 2874       var->okayToRealloc        = okayToRealloc;
 2875       var->decimation           = decimation;
 2876       var->numHits              = -1;  /* so first point gets logged */
 2877

 3219    * Abstract:
 3220    *	Called to log data for a log variable.

 3230   
 3231       int_T  offset        = 0;
 3232       char_T *currRealRow  = NULL;
 3233       char_T *currImagRow  = NULL;

 3235

 3237   
 3238       /* The following variables will be used for 
 3239          logging variable-size signals */
 3240       const  int_T  nDims = var->data.nDims;
 3241       const  int_T  *dims = var->data.dims;
 3242       const  void   * const *currDimsPtr = NULL;
 3243       const  int_T  *currDimsSizePtr = NULL;
 3244

 3246          logging "valueDimensions" field */

 3255           if (++var->numHits % var->decimation) continue;

 3259               if (var->okayToRealloc == 1) {
 3260                   rt_ReallocLogVar(var, isVarDims);
 3261               } else {
 3262                   /* Circular buffer */
 3263                   var->rowIdx = 0;
 3264                   ++(var->wrapped); /* increment the wrap around counter */
 3265               }
 3266           }
 3267   
 3268           if(isVarDims){
 3269               currDimsPtr = (const void * const *) var->valDims->currSigDims;
 3270               currDimsSizePtr = (const int_T*) var->valDims->currSigDimsSize;
 3271               logWidth_valDims = frameData ? 1 : var->valDims->nCols;
 3272               nRows_valDims = var->valDims->nRows;
 3273   
 3274               var->strides[0] = 1;
 3275               var->currStrides[0] = 1;
 3276   
 3277               for (k = 1; k < nDims; k++){
 3278                   int32_T currDimsVal=0;
 3279                   switch (currDimsSizePtr[k-1]) {
 3280                     case 1:
 3281                       currDimsVal = (**(((const uint8_T * const *) currDimsPtr)+(k-1)));
 3282                       break;
 3283                     case 2:
 3284                       currDimsVal = (**(((const uint16_T * const *) currDimsPtr)+(k-1)));
 3285                       break;
 3286                     case 4:
 3287                       currDimsVal = (**(((const uint32_T * const *) currDimsPtr)+(k-1)));
 3288                       break;
 3289                   }
 3290                   var->strides[k] = var->strides[k-1] * dims[k-1];
 3291                   var->currStrides[k] = var->currStrides[k-1] * currDimsVal;
 3292               }
 3293           }
 3294   
 3295           offset       = (int_T)(elSize * var->rowIdx * logWidth);
 3296           currRealRow  = ((char_T*) (var->data.re)) + offset;
 3297           currImagRow  = (var->data.complex) ?
 3298                          ((char_T*) (var->data.im)) + offset :  NULL;
 3299   
 3300           /* update logging data */
 3301           for (j = 0; j < logWidth; j++) {
 3302   
 3303               boolean_T inRange = true;
 3304               int idx = j;
 3305   
 3306               /* Check whether the currently logged value is in range or not.
 3307                  For fixed-size signal logging, always inRange = true; idx = j;
 3308                  For variable-size signal logging, use strides, coordinates
 3309                  and current strides to decide whether the currently logged
 3310                  data is in range or not and its location in the logging 
 3311                  matrix.
 3312                */
 3313               if(isVarDims){
 3314                   int rem = j;
 3315                   idx = 0;
 3316                   for(k = nDims-1; k>=0; k--){
 3317                       int32_T currDimsVal=0;
 3318                       switch (currDimsSizePtr[k]) {
 3319                         case 1:
 3320                           currDimsVal = (**(((const uint8_T * const *) currDimsPtr)+k));
 3321                           break;
 3322                         case 2:
 3323                           currDimsVal = (**(((const uint16_T * const *) currDimsPtr)+k));
 3324                           break;
 3325                         case 4:
 3326                           currDimsVal = (**(((const uint32_T * const *) currDimsPtr)+k));
 3327                           break;
 3328                       }

 3330                       if( var->coords[k] >= currDimsVal ){
 3331                           inRange = false;
 3332                           break;
 3333                       }
 3334                       rem = rem - var->coords[k] * var->strides[k];
 3335                   }
 3336                   if(inRange){
 3337                       idx = var->coords[0];
 3338                       for (k = 1; k < nDims; k++){
 3339                           idx += var->coords[k] * var->currStrides[k];
 3340                       }
 3341                   }
 3342               }
 3343               
 3344               if (!var->data.dataTypeConvertInfo.conversionNeeded) {
 3345                   /* NO  conversion needed
 3346                    */

 3350

 3438                               {
 3439                                   const real32_T *pInData = (const real32_T *)(cData + (i+frameSize*idx)* DpSize);

 3443                                       pInData = (const real32_T *)(cData + (i+frameSize*idx)* DpSize + DpSize/2);
