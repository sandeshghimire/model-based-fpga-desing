 1770     /* '<S11>:1:13' 'CastBeforeSum', true); */
 1771     /*  Overflow (V) */
 1772     /* '<S11>:1:16' overflow = bitget(in_flags, 2); */
 1773     rtb_out_flags = (uint8_T)((int32_T)((uint32_T)rtb_ALU_func >> 1) & 1);
 1774   
 1775     /*  Carry (C) */
 1776     /* '<S11>:1:19' c_out = bitget(in_flags, 4); */
 1777     rtb_AC_func = (uint8_T)((uint32_T)rtb_ALU_func >> 3);
 1778   
 1779     /* '<S11>:1:21' switch(uint8(select)) */
 1780     switch (rtb_shifter_sel) {
 1781      case 1U:
 1782       /* '<S11>:1:22' case 1, */
 1783       /*  shift left */
 1784       /*  affects C and V as well */
 1785       /* '<S11>:1:25' shift_out = bitsll(input, 1); */
 1786       rtb_AC_data = (int8_T)(diff << 1);
 1787   
 1788       /*  Carry (C) */
 1789       /* '<S11>:1:27' c_out = bitget(input, 8); */
 1790       rtb_AC_func = (uint8_T)((uint32_T)(uint8_T)diff >> 7);
 1791   
 1792       /*  Overflow (V) */
 1793       /* '<S11>:1:29' overflow = bitxor(bitget(input, 8), bitget(input, 7)); */
 1794       rtb_out_flags = (uint8_T)(((int32_T)((uint32_T)(uint8_T)diff >> 6) & 1) ^
 1795         (int32_T)((uint32_T)(uint8_T)diff >> 7));
 1796       break;
 1797   
 1798      case 2U:
 1799       /* '<S11>:1:30' case 2, */
 1800       /*  shift right */
 1801       /* '<S11>:1:32' shift_out = bitsra(input, 1); */
 1802       rtb_AC_data = (int8_T)(diff >> 1);
 1803       break;
 1804   
 1805      default:
 1806       /* '<S11>:1:33' otherwise */
 1807       /*  pass the input */
 1808       /* '<S11>:1:35' shift_out = fi(input, 1, 8, 0, hdl_fm); */
 1809       rtb_AC_data = diff;
 1810       break;
 1811     }
 1812   
 1813     /*  Negativity  (N) */
 1814     /* '<S11>:1:39' sign_bit = getmsb(shift_out); */
 1815     /*  Is Zero? (Z) */
 1816     /* '<S11>:1:42' is_zero = uint8(~all(shift_out)); */
 1817     /* '<S11>:1:43' zero_ufix1 = fi(is_zero, 0, 1, 0, hdl_fm); */
 1818     /*  Set [C, N, V, Z] in the flag register */
 1819     /* '<S11>:1:46' out_flags = bitconcat(bitconcat(bitconcat(c_out, sign_bit), overflow), zero_ufix1); */
 1820     hdlcodercpu_eml_B.out_flags = (uint8_T)(((rtb_AC_func << 1 | (int32_T)
 1821       ((uint32_T)(uint8_T)rtb_AC_data >> 7)) << 1 | rtb_out_flags) << 1 |
 1822       !(rtb_AC_data != 0));
 1823     hdlcodercpu_eml_B.shift_out = rtb_AC_data;
 1824   
 1825     /* End of MATLAB Function: '<S1>/Shifter (8-bit)' */
 1826   
 1827     /* MATLAB Function: '<S1>/output_enable' */
 1828     /* MATLAB Function 'CPU_Subsystem_8_bit/output_enable': '<S12>:1' */
 1829     /*  to the pin-out */
 1830     /*  HDL specific fimath */
 1831     /* '<S12>:1:5' hdl_fm = fimath(... */
 1832     /* '<S12>:1:6' 'RoundMode', 'floor',... */
 1833     /* '<S12>:1:7' 'OverflowMode', 'wrap',... */
 1834     /* '<S12>:1:8' 'ProductMode', 'FullPrecision', 'ProductWordLength', 32,... */
 1835     /* '<S12>:1:9' 'SumMode', 'FullPrecision', 'SumWordLength', 32,... */
 1836     /* '<S12>:1:10' 'CastBeforeSum', true); */
 1837     /* '<S12>:1:12' y = fi(0, 1, 8, 0, hdl_fm); */
 1838     hdlcodercpu_eml_B.y = 0;
 1839   
 1840     /* '<S12>:1:14' if(enable) */
 1841     if (rtb_print_data) {
 1842       /* '<S12>:1:15' y = fi(u, 1, 8, 0, hdl_fm); */
 1843       hdlcodercpu_eml_B.y = hdlcodercpu_eml_B.Switch1;
 1844     }
 1845   
 1846     /* End of MATLAB Function: '<S1>/output_enable' */
 1847   }

 1859     hdlcodercpu_eml_DW.UnitDelay3_DSTATE = hdlcodercpu_eml_B.out_flags;

 1861     /* Update for UnitDelay: '<S1>/Unit Delay 7' incorporates:
 1862      *  Constant: '<Root>/Master Reset'
 1863      */
 1864     hdlcodercpu_eml_DW.UnitDelay7_DSTATE = false;
 1865   
 1866     /* Update for UnitDelay: '<S1>/Unit Delay 1' */
 1867     hdlcodercpu_eml_DW.UnitDelay1_DSTATE = hdlcodercpu_eml_B.IR_out;

 1876     hdlcodercpu_eml_DW.we_d_DSTATE = hdlcodercpu_eml_B.DataTypeConversion;
 1877   
 1878     /* Update for UnitDelay: '<S6>/ram_data' */
 1879     hdlcodercpu_eml_DW.ram_data_DSTATE = hdlcodercpu_eml_B.select;
 1880   
 1881     /* Update for UnitDelay: '<S6>/ram' */
 1882     memcpy(&hdlcodercpu_eml_DW.ram_DSTATE[0], &hdlcodercpu_eml_B.switch_i[0],
 1883            sizeof(int8_T) << 8U);
 1884   
 1885     /* Update for UnitDelay: '<S1>/Unit Delay 4' */
 1886     hdlcodercpu_eml_DW.UnitDelay4_DSTATE = hdlcodercpu_eml_B.addr_out;
 1887   
 1888     /* Update absolute time for base rate */
 1889     /* The "clockTick0" counts the number of times the code of this task has
 1890      * been executed. The absolute time is the multiplication of "clockTick0"
 1891      * and "Timing.stepSize0". Size of "clockTick0" ensures timer will not
 1892      * overflow during the application lifespan selected.
 1893      * Timer of this task consists of two 32 bit unsigned integers.
 1894      * The two integers represent the low bits Timing.clockTick0 and the high bits
 1895      * Timing.clockTickH0. When the low bit overflows to 0, the high bits increment.
 1896      */
 1897     if (!(++hdlcodercpu_eml_M->Timing.clockTick0)) {
 1898       ++hdlcodercpu_eml_M->Timing.clockTickH0;
 1899     }
 1900   
 1901     hdlcodercpu_eml_M->Timing.t[0] = hdlcodercpu_eml_M->Timing.clockTick0 *
 1902       hdlcodercpu_eml_M->Timing.stepSize0 + hdlcodercpu_eml_M->Timing.clockTickH0 *
 1903       hdlcodercpu_eml_M->Timing.stepSize0 * 4294967296.0;
 1904   }
